MARC .mrk → .mrc


from pymarc import MARCMakerReader from js import Blob, document, URL def make_mrc(event=None): text = document.getElementById("mrk").value if not text.strip(): return reader = MARCMakerReader(text) data = b"".join(rec.as_marc() for rec in reader) blob = Blob.new([data], {"type": "application/marc"}) url = URL.createObjectURL(blob) link = document.createElement("a") link.href = url link.download = "records.mrc" document.body.appendChild(link) link.click() link.remove() URL.revokeObjectURL(url) document.getElementById("go").addEventListener("click", make_mrc)
