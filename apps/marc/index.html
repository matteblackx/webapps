import micropip
await micropip.install("pymarc")

from js import Blob, document, URL
from pymarc import MARCMakerReader
import io

def make_mrc(event=None):
    textarea = document.getElementById("mrk-input")
    log = document.getElementById("log")
    text = textarea.value if textarea is not None else ""

    if not text.strip():
        if log is not None:
            log.innerText = ">> No input detected in buffer."
        return

    try:
        reader = MARCMakerReader(io.StringIO(text))
        data = b"".join(rec.as_marc() for rec in reader)
        if not data:
            if log is not None:
                log.innerText = ">> Parser ran but produced zero records. Check .mrk formatting."
            return
    except Exception as e:
        if log is not None:
            log.innerText = f">> Error while parsing MARC: {e}"
        return

    blob = Blob.new([data], {"type": "application/octet-stream"})
    url = URL.createObjectURL(blob)
    a = document.createElement("a")
    a.href = url
    a.download = "records.mrc"
    document.body.appendChild(a)
    a.click()
    a.remove()
    URL.revokeObjectURL(url)

    if log is not None:
        log.innerText = ">> Build complete. Download initiated: records.mrc"

btn = document.getElementById("convert-btn")
if btn is not None:
    btn.addEventListener("click", make_mrc)
